# tkinter-рЃўрЃА рЃАрЃгрЃљрЃЋрЃџрЃћрЃЉрЃўрЃА рЃњрЃќрЃљрЃЏрЃЎрЃЋрЃџрЃћрЃЋрЃў

## ­ЪЊџ рЃерЃћрЃАрЃљрЃЋрЃљрЃџрЃў

рЃћрЃА рЃњрЃќрЃљрЃЏрЃЎрЃЋрЃџрЃћрЃЋрЃў рЃЊрЃљрЃњрЃћрЃ«рЃЏрЃљрЃарЃћрЃЉрЃљрЃЌ рЃўрЃАрЃгрЃљрЃЋрЃџрЃЮрЃЌ Python-рЃўрЃА tkinter рЃЉрЃўрЃЉрЃџрЃўрЃЮрЃЌрЃћрЃЎрЃљ GUI (рЃњрЃарЃљрЃцрЃўрЃЎрЃБрЃџрЃў рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў) рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃћрЃЉрЃўрЃА рЃерЃћрЃАрЃљрЃЦрЃЏрЃюрЃћрЃџрЃљрЃЊ.

---

## 1№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: tkinter-рЃўрЃА рЃАрЃљрЃцрЃБрЃФрЃЋрЃџрЃћрЃЉрЃў

### рЃарЃљ рЃљрЃарЃўрЃА tkinter?

tkinter рЃљрЃарЃўрЃА Python-рЃўрЃА рЃЕрЃљрЃерЃћрЃюрЃћрЃЉрЃБрЃџрЃў рЃЉрЃўрЃЉрЃџрЃўрЃЮрЃЌрЃћрЃЎрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃАрЃљрЃерЃБрЃљрЃџрЃћрЃЉрЃљрЃА рЃњрЃЋрЃљрЃФрЃџрЃћрЃЋрЃА рЃерЃћрЃЋрЃЦрЃЏрЃюрЃљрЃЌ рЃњрЃарЃљрЃцрЃўрЃЎрЃБрЃџрЃў рЃўрЃюрЃбрЃћрЃарЃцрЃћрЃўрЃАрЃў.

### рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃцрЃљрЃюрЃ»рЃљрЃарЃљ

```python
import tkinter as tk

# рЃЏрЃЌрЃљрЃЋрЃљрЃарЃў рЃцрЃљрЃюрЃ»рЃарЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ
root = tk.Tk()
root.title("рЃЕрЃћрЃЏрЃў рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃцрЃљрЃюрЃ»рЃљрЃарЃљ")
root.geometry("400x300")  # рЃАрЃўрЃњрЃљрЃюрЃћ x рЃАрЃўрЃЏрЃљрЃдрЃџрЃћ

# рЃцрЃљрЃюрЃ»рЃарЃўрЃА рЃњрЃљрЃерЃЋрЃћрЃЉрЃљ
root.mainloop()
```

**рЃњрЃљрЃюрЃЏрЃљрЃарЃбрЃћрЃЉрЃљ:**
- `tk.Tk()` - рЃЦрЃЏрЃюрЃўрЃА рЃЏрЃЌрЃљрЃЋрЃљрЃа рЃцрЃљрЃюрЃ»рЃљрЃарЃљрЃА
- `root.title()` - рЃљрЃДрЃћрЃюрЃћрЃЉрЃА рЃцрЃљрЃюрЃ»рЃарЃўрЃА рЃАрЃљрЃЌрЃљрЃБрЃарЃА
- `root.geometry()` - рЃљрЃДрЃћрЃюрЃћрЃЉрЃА рЃцрЃљрЃюрЃ»рЃарЃўрЃА рЃќрЃЮрЃЏрЃљрЃА
- `root.mainloop()` - рЃљрЃЦрЃбрЃўрЃБрЃарЃљрЃЊ рЃљрЃбрЃљрЃарЃћрЃЉрЃА рЃцрЃљрЃюрЃ»рЃљрЃарЃљрЃА (рЃБрЃюрЃЊрЃљ рЃўрЃДрЃЮрЃА рЃЉрЃЮрЃџрЃЮрЃерЃў)

## 2№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: Label (рЃГрЃЊрЃћ) рЃЋрЃўрЃ»рЃћрЃбрЃў

Label рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃњрЃљрЃЏрЃЮрЃАрЃљрЃЕрЃћрЃюрЃљрЃЊ.

```python
import tkinter as tk

root = tk.Tk()
root.title("Label рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў")
root.geometry("400x300")

# Label-рЃўрЃА рЃерЃћрЃЦрЃЏрЃюрЃљ
label1 = tk.Label(root, text="рЃћрЃА рЃљрЃарЃўрЃА рЃЕрЃћрЃЏрЃў рЃърЃўрЃарЃЋрЃћрЃџрЃў рЃГрЃЊрЃћ")
label1.pack()  # pack() рЃљрЃДрЃћрЃюрЃћрЃЉрЃА рЃЋрЃўрЃ»рЃћрЃбрЃА рЃцрЃљрЃюрЃ»рЃљрЃарЃљрЃерЃў

# Label рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃћрЃЉрЃўрЃЌ
label2 = tk.Label(
    root, 
    text="рЃцрЃћрЃарЃљрЃЊрЃў рЃбрЃћрЃЦрЃАрЃбрЃў",
    font=("Arial", 16),
    fg="blue",  # рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃцрЃћрЃарЃў
    bg="yellow"  # рЃцрЃЮрЃюрЃўрЃА рЃцрЃћрЃарЃў
)
label2.pack()

root.mainloop()
```

**рЃЏрЃюрЃўрЃерЃЋрЃюрЃћрЃџрЃЮрЃЋрЃљрЃюрЃў рЃЏрЃћрЃЌрЃЮрЃЊрЃћрЃЉрЃў:**
- `label.config(text="рЃљрЃ«рЃљрЃџрЃў рЃбрЃћрЃЦрЃАрЃбрЃў")` - рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃљ
- `label["text"] = "рЃљрЃ«рЃљрЃџрЃў рЃбрЃћрЃЦрЃАрЃбрЃў"` - рЃбрЃћрЃЦрЃАрЃбрЃўрЃА рЃерЃћрЃфрЃЋрЃџрЃўрЃА рЃАрЃ«рЃЋрЃљ рЃњрЃќрЃљ

## 3№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: Button (рЃдрЃўрЃџрЃљрЃЎрЃў) рЃЋрЃўрЃ»рЃћрЃбрЃў

Button рЃњрЃљрЃЏрЃЮрЃўрЃДрЃћрЃюрЃћрЃЉрЃљ рЃЏрЃЮрЃЏрЃ«рЃЏрЃљрЃарЃћрЃЉрЃџрЃўрЃА рЃЏрЃЮрЃЦрЃЏрЃћрЃЊрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА.

```python
import tkinter as tk

def button_clicked():
    print("рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ рЃЏрЃЮрЃ«рЃЊрЃљ!")
    label.config(text="рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃљ рЃЏрЃЮрЃ«рЃЊрЃљ!")

root = tk.Tk()
root.title("Button рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў")
root.geometry("400x300")

# Label
label = tk.Label(root, text="рЃЊрЃљрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЌ рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ")
label.pack(pady=20)

# Button
button = tk.Button(
    root,
    text="рЃЊрЃљрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЌ рЃљрЃЦ",
    command=button_clicked,  # рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃўрЃАрЃљрЃА
    bg="lightblue",
    fg="black",
    font=("Arial", 12)
)
button.pack()

root.mainloop()
```

**рЃњрЃљрЃюрЃЏрЃљрЃарЃбрЃћрЃЉрЃљ:**
- `command` - рЃцрЃБрЃюрЃЦрЃфрЃўрЃљ, рЃарЃЮрЃЏрЃћрЃџрЃўрЃф рЃњрЃљрЃћрЃерЃЋрЃћрЃЉрЃљ рЃдрЃўрЃџрЃљрЃЎрЃќрЃћ рЃЊрЃљрЃгрЃЎрЃљрЃърЃБрЃюрЃћрЃЉрЃўрЃАрЃљрЃА
- `pady` - рЃЋрЃћрЃарЃбрЃўрЃЎрЃљрЃџрЃБрЃарЃў рЃќрЃћрЃЊрЃюрЃљрЃДрЃЮрЃцрЃў (padding)

## 4№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: рЃЋрЃўрЃ»рЃћрЃбрЃћрЃЉрЃўрЃА рЃњрЃљрЃюрЃџрЃљрЃњрЃћрЃЉрЃљ

### pack() рЃЏрЃћрЃЌрЃЮрЃЊрЃў

```python
import tkinter as tk

root = tk.Tk()
root.title("рЃњрЃљрЃюрЃџрЃљрЃњрЃћрЃЉрЃљ")
root.geometry("400x300")

label1 = tk.Label(root, text="рЃќрЃћрЃЊрЃљ", bg="red")
label1.pack(pady=10)

label2 = tk.Label(root, text="рЃерЃБрЃљ", bg="green")
label2.pack(pady=10)

label3 = tk.Label(root, text="рЃЦрЃЋрЃћрЃЊрЃљ", bg="blue")
label3.pack(pady=10)

root.mainloop()
```

**pack() рЃърЃљрЃарЃљрЃЏрЃћрЃбрЃарЃћрЃЉрЃў:**
- `side="left"` рЃљрЃю `"right"` рЃљрЃю `"top"` рЃљрЃю `"bottom"`
- `padx=10` - рЃ░рЃЮрЃарЃўрЃќрЃЮрЃюрЃбрЃљрЃџрЃБрЃарЃў рЃќрЃћрЃЊрЃюрЃљрЃДрЃЮрЃцрЃў
- `pady=10` - рЃЋрЃћрЃарЃбрЃўрЃЎрЃљрЃџрЃБрЃарЃў рЃќрЃћрЃЊрЃюрЃљрЃДрЃЮрЃцрЃў

## 5№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: рЃЏрЃљрЃЌрЃћрЃЏрЃљрЃбрЃўрЃЎрЃБрЃарЃў рЃЮрЃърЃћрЃарЃљрЃфрЃўрЃћрЃЉрЃў

### 1. рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃўрЃА рЃ»рЃљрЃЏрЃў

```python
# рЃърЃўрЃарЃЋрЃћрЃџрЃў 10 рЃюрЃљрЃбрЃБрЃарЃљрЃџрЃБрЃарЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃ»рЃљрЃЏрЃў
sum_result = 0
for i in range(1, 11):  # 1-рЃЊрЃљрЃю 10-рЃЏрЃЊрЃћ
    sum_result += i
print(f"рЃ»рЃљрЃЏрЃў: {sum_result}")  # рЃерЃћрЃЊрЃћрЃњрЃў: 55
```

### 2. рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃБрЃџрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃЮ

```python
import math

# рЃърЃўрЃарЃЋрЃћрЃџрЃў 10 рЃюрЃљрЃбрЃБрЃарЃљрЃџрЃБрЃарЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃБрЃџрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃЮ
product = 1
for i in range(1, 11):
    product *= i

geometric_mean = product ** (1/10)  # 10-рЃћ рЃ«рЃљрЃарЃўрЃАрЃ«рЃўрЃА рЃцрЃћрЃАрЃЋрЃў
# рЃљрЃю math.pow(product, 1/10)
print(f"рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃБрЃџрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃЮ: {geometric_mean}")
```

### 3. рЃфрЃўрЃцрЃарЃЌрЃљ рЃюрЃљрЃЏрЃарЃљрЃЋрЃџрЃў

```python
def digit_product(number):
    product = 1
    for digit in str(number):
        if digit.isdigit():
            product *= int(digit)
    return product

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
result = digit_product(1234)  # 1 * 2 * 3 * 4 = 24
print(result)
```

### 4. рЃцрЃљрЃЦрЃбрЃЮрЃарЃўрЃљрЃџрЃў

```python
def factorial(n):
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

# 7-рЃўрЃА рЃцрЃљрЃЦрЃбрЃЮрЃарЃўрЃљрЃџрЃў
result = factorial(7)  # 7! = 5040
print(result)
```

### 5. рЃгрЃарЃцрЃўрЃЋрЃў рЃњрЃљрЃюрЃбрЃЮрЃџрЃћрЃЉрЃљ ax + b = 0

```python
def solve_linear(a, b):
    if a == 0:
        if b == 0:
            return "рЃБрЃАрЃљрЃАрЃарЃБрЃџрЃЮрЃЊ рЃЉрЃћрЃЋрЃарЃў рЃљрЃЏрЃЮрЃюрЃљрЃ«рЃАрЃюрЃў"
        else:
            return "рЃљрЃЏрЃЮрЃюрЃљрЃ«рЃАрЃюрЃў рЃљрЃа рЃљрЃарЃАрЃћрЃЉрЃЮрЃЉрЃА"
    else:
        x = -b / a
        return f"x = {x}"

# рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў
result = solve_linear(2, -4)  # 2x - 4 = 0 => x = 2
print(result)
```

### 6. рЃџрЃБрЃгрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃўрЃА рЃ»рЃљрЃЏрЃў рЃЊрЃљ рЃАрЃљрЃерЃБрЃљрЃџрЃЮ

```python
# 2-рЃЊрЃљрЃю 20-рЃЏрЃЊрЃћ рЃџрЃБрЃгрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў
even_numbers = []
for i in range(2, 21):
    if i % 2 == 0:  # рЃЌрЃБ 2-рЃќрЃћ рЃњрЃљрЃДрЃЮрЃцрЃўрЃАрЃљрЃА рЃюрЃљрЃерЃЌрЃў 0-рЃўрЃљ
        even_numbers.append(i)

sum_even = sum(even_numbers)
average = sum_even / len(even_numbers)

print(f"рЃџрЃБрЃгрЃў рЃарЃўрЃфрЃ«рЃЋрЃћрЃЉрЃў: {even_numbers}")
print(f"рЃ»рЃљрЃЏрЃў: {sum_even}")
print(f"рЃАрЃљрЃерЃБрЃљрЃџрЃЮ: {average}")
```

## 6№ИЈРЃБ рЃюрЃљрЃгрЃўрЃџрЃў: рЃАрЃарЃБрЃџрЃў рЃЏрЃљрЃњрЃљрЃџрЃўрЃЌрЃў - рЃЊрЃљрЃЋрЃљрЃџрЃћрЃЉрЃљ #1

```python
import tkinter as tk
import math

def calculate():
    # рЃърЃўрЃарЃЋрЃћрЃџрЃў 10 рЃюрЃљрЃбрЃБрЃарЃљрЃџрЃБрЃарЃў рЃарЃўрЃфрЃ«рЃЋрЃўрЃА рЃ»рЃљрЃЏрЃў
    sum_result = sum(range(1, 11))
    
    # рЃњрЃћрЃЮрЃЏрЃћрЃбрЃарЃўрЃБрЃџрЃў рЃАрЃљрЃерЃБрЃљрЃџрЃЮ
    product = 1
    for i in range(1, 11):
        product *= i
    geometric_mean = math.pow(product, 1/10)
```
